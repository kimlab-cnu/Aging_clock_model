# 최단 거리 재기
3. 최단 거리만 표시하는 커맨드 등록, 실행 → pep1–pep2 사이에서 가장 가까운 원자 쌍 1개만 라인으로 생성, 값 출력.
python
from pymol import cmd
import math

def mindist(sel1, sel2, obj_name="min_dist"):
    idx1 = cmd.index(sel1)
    idx2 = cmd.index(sel2)
    if not idx1 or not idx2:
        print("Selections are empty.")
        return
    best = None
    best_d = 1e9
    for m1,i1 in idx1:
        c1 = cmd.get_atom_coords(f"{m1}`{i1}", 1)
        for m2,i2 in idx2:
            c2 = cmd.get_atom_coords(f"{m2}`{i2}", 1)
            d = math.dist(c1, c2)
            if d < best_d:
                best_d = d
                best = (m1,i1,m2,i2)
    if best:
        cmd.delete(obj_name)
        cmd.distance(obj_name, f"{best[0]}`{best[1]}", f"{best[2]}`{best[3]}")
        print(f"Min distance = {best_d:.3f} Å between {best[0]}`{best[1]} and {best[2]}`{best[3]}")
cmd.extend("mindist", mindist)
python end

mindist pep1, pep2
